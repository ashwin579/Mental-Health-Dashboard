Step 1 — Deploy Apps Script in your spreadsheet
1. Open your Google Sheet
2. Click Extensions → Apps Script
3. Delete everything in the editor and paste the script below
4. Click Deploy → New deployment
5. Type: Web app · Execute as: Me · Who has access: Anyone
6. Click Deploy → copy the Web App URL → come to Step 2 
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const out = { sc: null, patients: null, meta: {} };

  // SC calls category mix tab
  const scSheet = ss.getSheetByName("SC calls category mix");
  if (scSheet) {
    out.sc = scSheet
      .getRange(1, 1, scSheet.getLastRow(), scSheet.getLastColumn())
      .getValues();
  }

  // All Patients Data tab
  const ptSheet = ss.getSheetByName("All Patients Data");
  if (ptSheet) {
    out.patients = ptSheet
      .getRange(1, 1, ptSheet.getLastRow(), ptSheet.getLastColumn())
      .getValues();
  }

  out.meta = {
    synced: new Date().toISOString(),
    spreadsheetName: ss.getName()
  };

  return ContentService
    .createTextOutput(JSON.stringify(out))
    .setMimeType(ContentService.MimeType.JSON);
}
